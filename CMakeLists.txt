CMAKE_MINIMUM_REQUIRED(VERSION 2.6)
PROJECT(CLIT)
SET(CMAKE_C_FLAGS "-ggdb -fPIC -pipe -Wall")
SET(CMAKE_LIBRARY_OUTPUT_DIRECTORY ${PROJECT_BINARY_DIR})

FIND_PACKAGE(PkgConfig)
pkg_check_modules(GTK3 REQUIRED gtk+-3.0)

ADD_EXECUTABLE(clit src/main.c src/ui/common.c src/ui/window.c)

INCLUDE_DIRECTORIES( ${GTK3_INCLUDE_DIRS} ${CLIT_SOURCE_DIR}/include)
TARGET_LINK_LIBRARIES(clit ${GTK3_LIBRARIES})
